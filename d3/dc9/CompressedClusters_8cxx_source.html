24 void CompressedClusters::Streamer(TBuffer& R__b)
25 {
26  // the custom streamer for CompressedClusters
27  if (R__b.IsReading()) {
28  R__b.ReadClassBuffer(CompressedClusters::Class(), this);
29  gsl::span flatdata{this->flatdata, this->flatdataSize};
30  CompressedClustersHelpers::restoreFrom(flatdata, *this);
31  } else {
32  std::vector<char> flatdata;
33  // member flatdata is nullptr unless it is an already extracted object which
34  // is streamed again. Overriding the existing flatbuffer is a potential
35  // memory leak
36  // Note: lots of cornercases here which are diffucult to catch with the design
37  // of CompClusters container (which is optimized to be used with GPU). Main concern
38  // is if an extracted object is not read-only and the counters are changed.
39  // TODO: we can add a consistency check whether the size of the flatbuffer
40  // and the pointers match the object.
41  bool isflat = this->flatdata != nullptr;
42  if (!isflat) {
43  CompressedClustersHelpers::flattenTo(flatdata, *this);
44  this->flatdataSize = flatdata.size();
45  this->flatdata = flatdata.data();
46  }
47  R__b.WriteClassBuffer(CompressedClusters::Class(), this);
48  if (!isflat) {
49  this->flatdataSize = 0;
50  this->flatdata = nullptr;
51  }
52  }
53 }
