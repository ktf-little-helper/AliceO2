76 template <typename H, typename... T>
77 constexpr auto make_originals_from_type()
78 {
79  using decayed = std::decay_t<H>;
80  if constexpr (sizeof...(T) == 0) {
81  if constexpr (is_type_with_originals_v<decayed>) {
82  return typename decayed::originals{};
83  } else if constexpr (is_type_with_originals_v<typename decayed::table_t>) {
84  return typename decayed::table_t::originals{};
85  } else {
86  return framework::pack<decayed>{};
87  }
88  } else if constexpr (is_type_with_originals_v<decayed>) {
89  return framework::concatenate_pack(typename decayed::originals{}, make_originals_from_type<T...>());
90  } else if constexpr (is_type_with_originals_v<typename decayed::table_t>) {
91  return framework::concatenate_pack(typename decayed::table_t::originals{}, make_originals_from_type<T...>());
92  } else {
93  return framework::concatenate_pack(framework::pack<decayed>{}, make_originals_from_type<T...>());

177 template <typename T, typename ChunkingPolicy = Chunked>
178 class ColumnIterator : ChunkingPolicy
179 {
180  static constexpr char SCALE_FACTOR = std::is_same_v<std::decay_t<T>, bool> ? 3 : 0;
181 
182  public:
187  ColumnIterator(BackendColumnType const* column)
188  : mColumn{column},
189  mCurrentPos{nullptr},
190  mFirstIndex{0},
191  mCurrentChunk{0}
192  {
193  auto chunks = framework::getBackendColumnPtrData<BackendColumnType>(mColumn);
194  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
195  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset();
196  mLast = mCurrent + array->length();
197  }
198 
199  ColumnIterator() = default;
200  ColumnIterator(ColumnIterator<T, ChunkingPolicy> const&) = default;
201  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy> const&) = default;
202 
203  ColumnIterator(ColumnIterator<T, ChunkingPolicy>&&) = default;
204  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy>&&) = default;
205 
207  void nextChunk() const
208  {
209  auto chunks = framework::getBackendColumnPtrData<BackendColumnType>(mColumn);
210  auto previousArray = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
211  mFirstIndex += previousArray->length();
212  mCurrentChunk++;
213  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
214  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
215  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
216  }
217 
218  void prevChunk() const
219  {
220  auto chunks = framework::getBackendColumnPtrData<BackendColumnType>(mColumn);
221  auto previousArray = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
222  mFirstIndex -= previousArray->length();
223  mCurrentChunk--;
224  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
225  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
226  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
227  }
228 
229  void moveToChunk(int chunk)
230  {
231  if (mCurrentChunk < chunk) {
232  while (mCurrentChunk != chunk) {
233  nextChunk();
234  }
235  } else {
236  while (mCurrentChunk != chunk) {
237  prevChunk();
238  }
239  }
240  }
241 
243  void moveToEnd()
244  {
245  auto chunks = framework::getBackendColumnPtrData<BackendColumnType>(mColumn);
246  mCurrentChunk = chunks->num_chunks() - 1;
247  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
248  assert(array.get());
249  mFirstIndex = mColumn->length() - array->length();
250  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
251  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
252  }
253 
254  decltype(auto) operator*() const
255  {
256  if constexpr (ChunkingPolicy::chunked) {
257  if (O2_BUILTIN_UNLIKELY(((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) >= mLast))) {
258  nextChunk();
259  }
260  }
261  if constexpr (std::is_same_v<bool, std::decay_t<T>>) {
262  // FIXME: notice that due to the type punning we cannot simply convert the
263  // masked char to a bool, because it's undefined behavior.
264  // FIXME: check if shifting the masked bit to the first position is better than != 0
265  return (*((char*)mCurrent + (*mCurrentPos >> SCALE_FACTOR)) & (1 << (*mCurrentPos & 0x7))) != 0;
266  } else {
267  return *(mCurrent + (*mCurrentPos >> SCALE_FACTOR));
268  }
269  }
270 
271  // Move to the chunk which containts element pos
272  ColumnIterator<T>& moveToPos()
273  {
274  // If we get outside range of the current chunk, go to the next.
275  if constexpr (ChunkingPolicy::chunked) {
276  while (O2_BUILTIN_UNLIKELY((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) >= mLast)) {
277  nextChunk();
278  }
279  }
280  return *this;
281  }
282 
283  // Move to the chunk which containts element pos
284  ColumnIterator<T>& checkNextChunk()
285  {
286  if constexpr (ChunkingPolicy::chunked) {
287  if (O2_BUILTIN_LIKELY((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) <= mLast)) {
288  return *this;
289  }
290  nextChunk();
291  }
292  return *this;
293  }
294 
295  mutable T const* mCurrent;
296  int64_t const* mCurrentPos;
297  mutable T const* mLast;
298  BackendColumnType const* mColumn;
299  mutable int mFirstIndex;
300  mutable int mCurrentChunk;
301 };
