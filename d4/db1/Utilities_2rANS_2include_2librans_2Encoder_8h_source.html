97 template <typename coder_t>
98 template <typename stream_IT, typename source_IT>
99 const stream_IT Encoder<coder_t>::Encoder::process(const stream_IT outputBegin, const stream_IT outputEnd, const source_IT inputBegin, const source_IT inputEnd) const
100 {
101 
102  typedef typename std::iterator_traits<stream_IT>::value_type stream_t;
103  using ransCoder = Coder<coder_t, stream_t>;
104 
105  State<coder_t> rans0, rans1;
106  ransCoder::encInit(&rans0);
107  ransCoder::encInit(&rans1);
108 
109  stream_t* ptr = &(*outputEnd);
110  source_IT inputIT = inputEnd;
111 
112  const auto inputBufferSize = inputEnd - inputBegin;
113 
114  // odd number of bytes?
115  if (inputBufferSize & 1) {
116  const coder_t s = *(--inputIT);
117  ransCoder::encPutSymbol(&rans0, &ptr, &(*mSymbolTable)[s],
118  mProbabilityBits);
119  }
120 
121  while (inputIT > inputBegin) { // NB: working in reverse!
122  const coder_t s1 = *(--inputIT);
123  const coder_t s0 = *(--inputIT);
124  ransCoder::encPutSymbol(&rans1, &ptr, &(*mSymbolTable)[s1],
125  mProbabilityBits);
126  ransCoder::encPutSymbol(&rans0, &ptr, &(*mSymbolTable)[s0],
127  mProbabilityBits);
128  }
129  ransCoder::encFlush(&rans1, &ptr);
130  ransCoder::encFlush(&rans0, &ptr);
131 
132  return outputBegin + std::distance(&(*outputBegin), ptr);
133 };
