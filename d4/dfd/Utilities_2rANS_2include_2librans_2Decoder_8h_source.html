80 void Decoder<coder_t, source_T>::process(const source_IT outputBegin, const stream_IT inputBegin, size_t numSymbols) const
81 {
82  static_assert(std::is_same<typename std::iterator_traits<source_IT>::value_type, source_T>::value);
83 
84  typedef typename std::iterator_traits<stream_IT>::value_type stream_t;
85  using ransDecoder = Coder<coder_t, stream_t>;
86 
87  State<coder_t> rans0, rans1;
88  stream_t* ptr = &(*inputBegin);
89  source_IT it = outputBegin;
90  ransDecoder::decInit(&rans0, &ptr);
91  ransDecoder::decInit(&rans1, &ptr);
92 
93  for (size_t i = 0; i < (numSymbols & ~1); i += 2) {
94  const stream_t s0 =
95  (*mReverseLUT)[ransDecoder::decGet(&rans0, mProbabilityBits)];
96  const stream_t s1 =
97  (*mReverseLUT)[ransDecoder::decGet(&rans1, mProbabilityBits)];
98  *it++ = s0;
99  *it++ = s1;
100  ransDecoder::decAdvanceSymbolStep(&rans0, &(*mSymbolTable)[s0],
101  mProbabilityBits);
102  ransDecoder::decAdvanceSymbolStep(&rans1, &(*mSymbolTable)[s1],
103  mProbabilityBits);
104  ransDecoder::decRenorm(&rans0, &ptr);
105  ransDecoder::decRenorm(&rans1, &ptr);
106  }
107 
108  // last byte, if number of bytes was odd
109  if (numSymbols & 1) {
110  const stream_t s0 =
111  (*mReverseLUT)[ransDecoder::decGet(&rans0, mProbabilityBits)];
112  *it = s0;
113  ransDecoder::decAdvanceSymbol(&rans0, &ptr, &(*mSymbolTable)[s0],
114  mProbabilityBits);
115  }
116 }
